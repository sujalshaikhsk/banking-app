import lombok.Setter;

@Entity
@Table(name = "banks")
@Setter
@Getter
public class Bank implements Serializable {

	private static final long serialVersionUID = 1L;

	@Id
	private String ifscCode;
	private String bankName;
	private String branchName;

}
